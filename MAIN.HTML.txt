                      <button onClick={() => handleNavClick('home')} className="inline-flex items-center px-6 py-3 border border-slate-300 shadow-sm text-base font-medium rounded-md text-slate-700 bg-white hover:bg-slate-50 focus:outline-none">
                        <Home className="w-5 h-5 mr-2 -ml-1 text-slate-500" />
                        返回首頁
                      </button>
                   </div>
                </div>
              </div>
            );
          };

          const MembersView = () => {
            const masterYears = [...new Set(membersData
              .filter(m => m.type === 'master' && m.classYear)
              .map(m => m.classYear)
            )].sort((a, b) => b - a);

            const filteredMembers = membersData.filter(m => {
               if (memberFilter === 'all') return true;
               if (!['all', 'phd', 'alumni'].includes(memberFilter)) {
                 return m.classYear === memberFilter;
               }
               return m.type === memberFilter;
            });

            return (
              <div className="min-h-screen bg-slate-50 pb-20">
                <div className="bg-slate-900 text-white pt-24 pb-12 px-4">
                  <div className="max-w-7xl mx-auto px-4">
                    <button 
                      onClick={() => handleNavClick('home')}
                      className="flex items-center text-blue-300 hover:text-white mb-6 transition-colors group"
                    >
                      <ArrowLeft className="w-5 h-5 mr-2 group-hover:-translate-x-1 transition-transform" />
                      返回首頁
                    </button>
                    <h1 className="text-3xl md:text-4xl font-bold flex items-center">
                      <Users className="w-10 h-10 mr-4 text-blue-500" />
                      實驗室成員
                    </h1>
                    <p className="mt-4 text-slate-400 text-lg max-w-2xl">
                      認識我們充滿活力與熱情的研究團隊，我們來自不同背景，共同致力於智慧製造技術的創新。
                    </p>
                  </div>
                </div>

                <div className="sticky top-16 z-30 bg-white border-b border-slate-200 shadow-sm">
                  <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 h-16 flex items-center space-x-4 overflow-x-auto">
                     <Filter className="w-5 h-5 text-slate-400 flex-shrink-0" />
                     <div className="flex space-x-2">
                       <button onClick={() => setMemberFilter('all')} className={`px-4 py-1.5 rounded-full text-sm font-medium transition-colors whitespace-nowrap ${memberFilter === 'all' ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-600 hover:bg-slate-200'}`}>全部成員</button>
                       <button onClick={() => setMemberFilter('phd')} className={`px-4 py-1.5 rounded-full text-sm font-medium transition-colors whitespace-nowrap ${memberFilter === 'phd' ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-600 hover:bg-slate-200'}`}>博士生</button>
                       
                       {masterYears.map(year => (
                         <button 
                           key={year}
                           onClick={() => setMemberFilter(year)} 
                           className={`px-4 py-1.5 rounded-full text-sm font-medium transition-colors whitespace-nowrap ${memberFilter === year ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-600 hover:bg-slate-200'}`}
                         >
                           {year}級碩士
                         </button>
                       ))}

                       <button onClick={() => setMemberFilter('alumni')} className={`px-4 py-1.5 rounded-full text-sm font-medium transition-colors whitespace-nowrap ${memberFilter === 'alumni' ? 'bg-blue-600 text-white' : 'bg-slate-100 text-slate-600 hover:bg-slate-200'}`}>畢業校友</button>
                     </div>
                  </div>
                </div>

                <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-12">
                  {memberFilter === 'all' && (
                     <div className="mb-16">
                       <h2 className="text-2xl font-bold text-slate-900 mb-8 border-l-4 border-blue-600 pl-4">指導教授</h2>
                       <div className="bg-white rounded-2xl p-8 border border-slate-200 shadow-sm flex flex-col md:flex-row items-center md:items-start gap-8">
                          <div className="w-48 h-48 bg-slate-200 rounded-full flex-shrink-0 overflow-hidden flex items-center justify-center border-4 border-slate-50 shadow-inner">
                            <User className="w-24 h-24 text-slate-400" />
                          </div>
                          <div>
                            <h3 className="text-3xl font-bold text-slate-900 mb-2">{labInfo.pi}</h3>
                            <p className="text-blue-700 font-medium text-lg mb-4">{labInfo.piTitle}</p>
                            <p className="text-slate-600 mb-6 leading-relaxed max-w-3xl">
                              林教授畢業於國立成功大學機械工程博士，現任勤益科大機械系副系主任。
                              專長涵蓋精密製造、磨潤學及智慧監控系統。
                              致力於將學術研究轉化為實際產業應用，帶領團隊進行前瞻性技術開發。
                            </p>
                            <div className="flex flex-wrap gap-4">
                              <a href={`mailto:${labInfo.email}`} className="flex items-center text-slate-600 hover:text-blue-700 transition-colors bg-slate-50 px-4 py-2 rounded-lg">
                                <Mail className="w-4 h-4 mr-2"/> {labInfo.email}
                              </a>
                              <div className="flex items-center text-slate-600 bg-slate-50 px-4 py-2 rounded-lg">
                                <MapPin className="w-4 h-4 mr-2"/> {labInfo.location}
                              </div>
                            </div>
                          </div>
                       </div>
                     </div>
                  )}

                  <h2 className="text-2xl font-bold text-slate-900 mb-8 border-l-4 border-blue-600 pl-4">
                    {memberFilter === 'all' ? '研究團隊' : 
                     memberFilter === 'phd' ? '博士生' : 
                     memberFilter === 'alumni' ? '畢業校友' : 
                     `${memberFilter}級 碩士生`}
                  </h2>
                  <div className="grid gap-8 md:grid-cols-2 lg:grid-cols-3">
                    {filteredMembers.map((member) => (
                      <div key={member.id} className="bg-white rounded-xl overflow-hidden shadow-sm border border-slate-200 hover:shadow-lg transition-all group flex flex-col h-full">
                        <div className="p-6 text-center border-b border-slate-100 bg-slate-50/50">
                           <div className="w-32 h-32 bg-white rounded-full mx-auto mb-4 flex items-center justify-center text-slate-300 border-4 border-white shadow-sm overflow-hidden relative">
                              <User className="w-16 h-16" />
                           </div>
                           <h3 className="text-xl font-bold text-slate-900">{member.name}</h3>
                           <p className="text-slate-500 font-medium">{member.year}</p>
                           <p className="text-xs text-blue-600 mt-1 uppercase tracking-wide">{member.title}</p>
                        </div>
                        
                        <div className="p-6 flex-grow flex flex-col">
                           <div className="mb-4">
                             <h4 className="text-sm font-bold text-slate-900 mb-2 flex items-center">
                               <Lightbulb className="w-4 h-4 mr-1.5 text-blue-500" /> 主要研究
                             </h4>
                             <p className="text-sm text-slate-600 leading-relaxed">
                               {member.researchDetail}
                             </p>
                           </div>
                           
                           {member.awards && member.awards.length > 0 && (
                             <div className="mt-auto pt-4 border-t border-slate-100">
                                <h4 className="text-sm font-bold text-slate-900 mb-2 flex items-center">
                                  <Trophy className="w-4 h-4 mr-1.5 text-yellow-500" /> 獲獎紀錄
                                </h4>
                                <ul className="text-sm text-slate-600 space-y-1">
                                  {member.awards.map((award, idx) => (
                                    <li key={idx} className="flex items-start">
                                      <span className="mr-1.5 text-slate-400">•</span>
                                      <span>{award}</span>
                                    </li>
                                  ))}
                                </ul>
                             </div>
                           )}
                        </div>
                      </div>
                    ))}
                  </div>
                  
                  {filteredMembers.length === 0 && (
                     <div className="text-center py-12 text-slate-400 bg-slate-50 rounded-lg border border-dashed border-slate-300">
                       目前尚無此類別成員資料。
                     </div>
                  )}

                  <div className="mt-12 text-center">
                     <button onClick={() => handleNavClick('home')} className="inline-flex items-center px-6 py-3 border border-slate-300 shadow-sm text-base font-medium rounded-md text-slate-700 bg-white hover:bg-slate-50 focus:outline-none">
                       <Home className="w-5 h-5 mr-2 -ml-1 text-slate-500" />
                       返回首頁
                     </button>
                  </div>
                </div>
              </div>
            );
          };

          return (
            <div className="min-h-screen bg-slate-50 font-sans text-slate-800">
              <Navigation />
              {currentPage === 'home' && <HomeView />}
              {currentPage === 'projects' && <ProjectsView />}
              {currentPage === 'members' && <MembersView />}
              <Footer />
            </div>
          );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<LabWebsite />);
    </script>
</body>
